        function useLocalStorage(key, initialValue) {
          const [storedValue, setStoredValue] = useState(() => {
            try {
              const item = window.localStorage.getItem(key);
              return item ? JSON.parse(item) : initialValue;
            } catch (error) { return initialValue; }
          });
          const setValue = (value) => {
            try {
              const valueToStore = value instanceof Function ? value(storedValue) : value;
              setStoredValue(valueToStore);
              window.localStorage.setItem(key, JSON.stringify(valueToStore));
            } catch (error) { console.error(error); }
          };
          return [storedValue, setValue];
        }

            if (!GAS_URL || GAS_URL === "MASUKKAN_URL_WEB_APP_ANDA_DISINI") {
                setDbError(true); throw new Error("URL API belum dikonfigurasi");
            }
            try {
                const response = await fetch(GAS_URL, {
                    method: 'POST', headers: { 'Content-Type': 'text/plain;charset=utf-8' },
                    body: JSON.stringify({ action, ...payload })

                return await response.json();
            } catch (error) { throw error; }

             }
             setIsFetching(true);
             try {
                const res = await fetch(`${GAS_URL}?action=READ_ALL`);
                const data = await res.json();
                setUsers(data.users || []); setStudents(data.students || []); setViolations(data.violations || []);
                setLeaves(data.leaves || []); setEvaluations(data.evaluations || []);
             } catch (err) {
                showToast("Gagal memuat data dari cloud. Menampilkan data lokal jika ada.");
             } finally { setIsFetching(false); }
          };

          useEffect(() => { if(role) loadDataFromBackend(); }, [role]);

          const showToast = (message) => { setToast(message); setTimeout(() => setToast(null), 3000); };
          const showConfirm = (message, onConfirm) => { setConfirmModal({ isOpen: true, message, onConfirm }); };
          const handleConfirmAction = () => { if (confirmModal.onConfirm) confirmModal.onConfirm(); handleCloseConfirm(); };

            setIsFetching(true);
            try {
              const res = await apiCall('LOGIN', { data: { role: selectedRole, password } });
              if (res.success) {
                setRole(selectedRole); setActiveMenu('dashboard');
                showToast(`Autentikasi berhasil. Halo, Tim ${selectedRole}!`);
              } else { showToast(res.error || 'Kredensial tidak valid!'); }
            } catch (e) { showToast('Koneksi terputus. Gagal memverifikasi.'); } 
            finally { setIsFetching(false); }
          };
          
          const handleLogout = () => setRole(null);

          const handleStudentOp = async (action, data) => {
              showToast('Menyinkronkan ke Cloud...');
              try { await apiCall(action, { data }); showToast('Sinkronisasi sukses.'); } 
              catch (e) { showToast('Gagal sinkronisasi. Periksa koneksi Anda.'); }
          };

          const addNewStudent = (nisn, name, gender, className, room) => {
            const newStudent = { id: `S${Date.now()}`, nisn, name, gender, class: className, room, points: 0 };
            setStudents([newStudent, ...students]); handleStudentOp('ADD_STUDENT', newStudent);

            const formatted = newStudentsList.map(s => ({

            }));
            setStudents([...formatted, ...students]);
            showToast('Menyinkronkan batch data...');
            try { for(let st of formatted) await apiCall('ADD_STUDENT', { data: st }); showToast('Batch sukses tersimpan!'); } 
            catch (e) { showToast('Beberapa data mungkin gagal disinkronkan.'); }
          };

          const updateStudent = (id, updatedData) => {
            const currentObj = students.find(s => s.id === id);
            const newObj = { ...currentObj, ...updatedData };
            setStudents(students.map(s => s.id === id ? newObj : s)); handleStudentOp('UPDATE_STUDENT', newObj);
          };

          const deleteStudent = (id) => {
            showConfirm('Hapus data siswa ini beserta riwayatnya secara permanen?', () => {
              setStudents(students.filter(s => s.id !== id));
              setViolations(violations.filter(v => v.studentId !== id));
              setLeaves(leaves.filter(l => l.studentId !== id));
              setEvaluations(evaluations.filter(e => e.studentId !== id));
              handleStudentOp('DELETE_STUDENT', {id});
            });
          };

          const addViolation = (studentId, description, points, photoData = null) => {
            const pts = parseInt(points);
            const newV = { id: `V${Date.now()}`, studentId, description, points: pts, date: new Date().toISOString(), photoData };
            
            const tempV = { ...newV, photoUrl: photoData ? `data:${photoData.mimeType};base64,${photoData.base64}` : '' };
            setViolations([tempV, ...violations]);
            
            const upStudent = { ...students.find(s => s.id === studentId) }; upStudent.points += pts;
            setStudents(students.map(s => s.id === studentId ? upStudent : s));
            
            try { apiCall('ADD_VIOLATION', { data: newV }); apiCall('UPDATE_STUDENT', { data: upStudent }); showToast('Sanksi masuk Cloud.'); } catch (e) { showToast('Gagal sinkronisasi laporan.'); }

            let updatedObj = null;

              return l;
            }));
            if(updatedObj) handleStudentOp('UPDATE_LEAVE', updatedObj);

                    onStudentClick={(id) => setSelectedStudentId(id)} onResetData={resetAllData} onBackupData={handleBackupData} onRestoreData={handleRestoreData} showToast={showToast}

            const stuL = leaves.filter(l => l.studentId === studentId).sort((a,b) => new Date(b.history[0].time)-new Date(a.history[0].time));
